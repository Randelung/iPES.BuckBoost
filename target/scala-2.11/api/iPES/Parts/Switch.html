case class Switch(start: Vector2D, end: Vector2D, animator: Animator, inputHandler: InputHandler = null, closed: Boolean = true, length: Double = 20) extends Animatable with Product with Serializable

end

animator

Animator, in which to register itself

inputHandler

Input handler, in which to register itself. Can be omitted, but is not recommended, since a switch is an active component.

closed

Initial (and current) state of switch. Default is closed/conducting.

length

Length of switch piece itself. Default is 20 pixels.
